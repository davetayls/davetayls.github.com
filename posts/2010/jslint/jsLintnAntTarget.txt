	<target name="jsLintValidation" description="validates JS files">
		<foreach item="File" property="jsfile">
			<in>
				<items>
					<include name="${root}\js\*.js" />
				</items>
			</in>
			<do>
				<exec program="${jslint}\jslint.bat" commandline="${jsfile}" />
			</do>
		</foreach>
	</target>
